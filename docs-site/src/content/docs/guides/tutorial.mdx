import { FileTree, Aside, Steps } from '@astrojs/starlight/components';

fluxe scaffold một **project workspace member** tách `backend/` (API + data theo module) và
`frontend/` (cells theo feature). Bạn chỉ viết trong project; engine đến từ `@nmvuong92/fluxe`.

## 1. Scaffold — `fx init`

<Steps>

1. Tạo project (đặt tên tuỳ ý — `app`, `app2`…):

   ```bash
   fx init app --server=fastify --driver=memory
   ```

   Cờ: `--driver=memory|sqlite|postgres` · `--server=express|fastify` · `--auth`.

2. Cấu trúc sinh ra:

   <FileTree>
   - app/
     - package.json · tsconfig.json (workspace member tự chứa)
     - backend/
       - db.ts driver data (`makeDb`)
       - contract.ts · app.ts (`makeApp` = `createApp`) · server.ts (Fastify)
       - modules/todos/ **feature = local plugin**
         - todos.contract.ts · todos.service.ts · todos.resolvers.ts · todos.plugin.ts
       - tests/ unit/ · e2e/ · helpers/
     - frontend/
       - features/home/ · greet/ · todos/  (`<name>.cell.tsx` + `<name>.view.tsx`)
       - layouts/ · i18n.ts · api.ts · profiles.ts · client.tsx
   </FileTree>

3. Chạy dev:

   ```bash
   npm run dev        # sync + resolve + bundle + server (watch)
   ```

</Steps>

<Aside type="note" title="registry là file SINH TỰ ĐỘNG">
`fx sync` quét `frontend/features/**` sinh `frontend/registry.ts` (cells) + `frontend/views.ts`
(views client). `dev`/`resolve` tự gọi `sync` → **thêm trang = tạo file, không khai báo tay**. Cả
hai file được gitignore.
</Aside>

## 2. Thêm một trang — `fx new`

```bash
fx new blog/post            # sinh frontend/features/blog/post.cell.tsx + post.view.tsx (island)
fx new about --static       # 0-JS
```

**`<name>.view.tsx`** — giao diện thuần (`export default` để client bundle nhận):

```tsx
// frontend/features/blog/post.view.tsx
export interface PostData { title: string }
export function Post({ data }: { data: PostData }) {
  return <div className="card"><h1>{data.title}</h1></div>;
}
export default Post;
```

**`<name>.cell.tsx`** — route + loader (server), gắn view. **cell.id === basename**:

```tsx
// frontend/features/blog/post.cell.tsx
import { defineCell } from "@nmvuong92/fluxe";
import { Post } from "./post.view";
export default defineCell({
  id: "post",
  route: "/post",
  layout: "site",
  async loader() { return { title: "Bài viết" }; },
  head: (d) => ({ title: d.title }),
  view: Post,
});
```

`fx new` tự chạy `sync` → cell đăng ký ngay. Mở `http://localhost:5180/post`.

## 3. Thêm một backend module (contract + resolver)

Mỗi feature backend = **một local plugin** gói contract + resolvers.

```ts
// backend/modules/todos/todos.contract.ts — nguồn sự thật (Zod, 0 codegen)
import { f } from "@nmvuong92/fluxe";
const Todo = f.object({ id: f.string, title: f.string, done: f.bool });
export const todosContract = f.contract({
  listTodos: f.query(Todo.array()),
  addTodo: f.mutation({ title: f.string }, Todo),
  onTodos: f.subscription(Todo.array()),
});
```

```ts
// backend/modules/todos/todos.resolvers.ts — plain function, ctx tường minh (0 decorator/DI magic)
import type { Resolvers } from "@nmvuong92/fluxe";
import type { TodoStore } from "@backend/db";
import { todosContract } from "./todos.contract.ts";
import { makeTodosService } from "./todos.service.ts";
export function makeTodosResolvers(store: TodoStore): Resolvers<typeof todosContract> {
  const svc = makeTodosService(store);
  return {
    listTodos: () => svc.list(),
    addTodo: async ({ title }, ctx) => { const t = await svc.add(title); ctx.publish("onTodos", await svc.list()); return t; },
  };
}
```

```ts
// backend/modules/todos/todos.plugin.ts — definePlugin gói lại
import { definePlugin } from "@nmvuong92/fluxe";
import { todosContract } from "./todos.contract.ts";
import { makeTodosResolvers } from "./todos.resolvers.ts";
export function todosPlugin(store) {
  return definePlugin({ name: "@app/todos", contract: todosContract, resolvers: makeTodosResolvers(store) });
}
```

Ghép trong `backend/app.ts` — scale = thêm module vào `plugins`:

```ts
export async function makeApp() {
  const store = makeDb();
  return createApp({ manifest, cells, layouts, i18n, plugins: [todosPlugin(store)], backend: store });
}
```

Frontend gọi op qua hook typed (0 schema xuống browser):

```tsx
import { api } from "../../api";   // createHooks<typeof contract>()
const q = api.listTodos.useQuery({ initial: data.todos });
const form = api.addTodo.useForm({ invalidates: ["listTodos"] });
api.onTodos.useSubscription(() => q.refetch());   // realtime SSE
```

## 4. Test — vùng test riêng

```bash
npm test           # unit + e2e (backend/tests/)
```

```ts
// backend/tests/e2e/todos.e2e.test.ts — boot app thật, HTTP thật
import { startTestServer } from "../helpers/make-test-app.ts";
test("addTodo rồi listTodos thấy todo", async () => {
  const { port, close } = await startTestServer();
  await fetch(`http://127.0.0.1:${port}/__rpc/addTodo`, { method: "POST", headers: { "content-type": "application/json" }, body: JSON.stringify({ title: "học fluxe" }) });
  const todos = await (await fetch(`http://127.0.0.1:${port}/__rpc/listTodos`, { method: "POST" })).json();
  assert.ok(todos.some((t) => t.title === "học fluxe"));
  await close();
});
```
